// Fetch and transform data
async function loadMetrics(opts) {
  const res = await fetch(
    `/api/v3/metrics`,
    { headers: { 'X-Api-Key': opts.key } }
  );
  const data = await res.json();
  return data.map(d => ({
    name: d.label,
    value: d.count * 1.15
  }));
}
